.clear-fix {
  &::after {
    content: "";
    display: table;
    clear: both;
  }
}

.border-top-1px,
.border-right-1px,
.border-bottom-1px,
.border-left-1px {
  position: relative;

  &::before,
  &::after {
    content: "";
    display: block;
    position: absolute;
    transform-origin: 0 0;
  }
}

$color-row-line: #ebebeb;
$color-col-line: #ebebeb;

.border-top-1px {
  &::before {
    border-top: 1px solid $color-row-line;
    left: 0;
    top: 0;
    width: 100%;
    transform-origin: 0 top;
  }
}

.border-right-1px {
  &::after {
    border-right: 1px solid $color-col-line;
    top: 0;
    right: 0;
    height: 100%;
    transform-origin: right 0;
  }
}

.border-bottom-1px {
  &::after {
    border-bottom: 1px solid $color-row-line;
    left: 0;
    bottom: 0;
    width: 100%;
    transform-origin: 0 bottom;
